# v1.1 Security Hardening Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add gRPC mTLS with hub-issued short-lived certificates and httpOnly cookie auth with CSRF protection.

**Architecture:** Hub generates an ECDSA P-256 CA on first boot. Agents generate keypairs locally and send CSRs during registration; hub signs them into 12h certs. Cert renewal happens in-stream. For web auth, JWT tokens move from localStorage to httpOnly cookies with a double-submit CSRF pattern.

**Tech Stack:** Go `crypto/ecdsa`, `crypto/x509`, `crypto/tls`; Protocol Buffers; React; SQLite migration

**Spec:** `docs/superpowers/specs/2026-03-28-v1.1-security-hardening-design.md`

package ca

import (
	"crypto/ecdsa"
	"crypto/elliptic"
	"crypto/rand"
	"crypto/x509"
	"crypto/x509/pkix"
	"math/big"
	"time"
)

// GenerateCA creates a new ECDSA P-256 CA keypair with a self-signed certificate.
func GenerateCA() (*x509.Certificate, *ecdsa.PrivateKey, error) {
	key, err := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
	if err != nil {
		return nil, nil, err
	}

	serial, err := rand.Int(rand.Reader, new(big.Int).Lsh(big.NewInt(1), 128))
	if err != nil {
		return nil, nil, err
	}

	template := &x509.Certificate{
		SerialNumber: serial,
		Subject:      pkix.Name{CommonName: "AeroDocs CA"},
		NotBefore:    time.Now().Add(-1 * time.Minute),
		NotAfter:     time.Now().Add(10 * 365 * 24 * time.Hour),
		KeyUsage:     x509.KeyUsageCertSign | x509.KeyUsageCRLSign,
		BasicConstraintsValid: true,
		IsCA:                  true,
		MaxPathLen:            0,
	}

	certDER, err := x509.CreateCertificate(rand.Reader, template, template, &key.PublicKey, key)
	if err != nil {
		return nil, nil, err
	}

	cert, err := x509.ParseCertificate(certDER)
	if err != nil {
		return nil, nil, err
	}

	return cert, key, nil
}
```

// SignCSR signs a DER-encoded CSR with the CA, producing a client certificate
// with the given serverID as CN and the specified validity duration.
func SignCSR(caCert *x509.Certificate, caKey *ecdsa.PrivateKey, csrDER []byte, serverID string, validity time.Duration) (*x509.Certificate, error) {
	csr, err := x509.ParseCertificateRequest(csrDER)
	if err != nil {
		return nil, fmt.Errorf("parse CSR: %w", err)
	}
	if err := csr.CheckSignature(); err != nil {
		return nil, fmt.Errorf("invalid CSR signature: %w", err)
	}

	serial, err := rand.Int(rand.Reader, new(big.Int).Lsh(big.NewInt(1), 128))
	if err != nil {
		return nil, err
	}

	template := &x509.Certificate{
		SerialNumber: serial,
		Subject:      pkix.Name{CommonName: serverID},
		NotBefore:    time.Now().Add(-1 * time.Minute),
		NotAfter:     time.Now().Add(validity),
		KeyUsage:     x509.KeyUsageDigitalSignature,
		ExtKeyUsage:  []x509.ExtKeyUsage{x509.ExtKeyUsageClientAuth},
	}

	certDER, err := x509.CreateCertificate(rand.Reader, template, caCert, csr.PublicKey, caKey)
	if err != nil {
		return nil, fmt.Errorf("sign certificate: %w", err)
	}

	return x509.ParseCertificate(certDER)
}
```

import (
	"database/sql"

func (s *Store) SaveCA(certDER, keyEncrypted []byte) error {
	_, err := s.db.Exec(
		`INSERT OR REPLACE INTO ca_config (id, ca_cert, ca_key_encrypted) VALUES ('default', ?, ?)`,
		certDER, keyEncrypted,
	)
	return err
}

func (s *Store) LoadCA() (certDER, keyEncrypted []byte, err error) {
	err = s.db.QueryRow(`SELECT ca_cert, ca_key_encrypted FROM ca_config WHERE id = 'default'`).
		Scan(&certDER, &keyEncrypted)
	if err == sql.ErrNoRows {
		return nil, nil, fmt.Errorf("CA not initialized")
	}
	return certDER, keyEncrypted, err
}

	if h.ca == nil {
		stream.Send(&pb.HubMessage{Payload: &pb.HubMessage_CertRenewResponse{

		}})

			CertRenewResponse: &pb.CertRenewResponse{Error: err.Error()},
		}})
		return
	}

	stream.Send(&pb.HubMessage{Payload: &pb.HubMessage_CertRenewResponse{
		CertRenewResponse: &pb.CertRenewResponse{
			ClientCert: cert,
			CaCert:     h.ca.CACertDER(),
		},
	}})
}

```go
package certs

import (
	"crypto/ecdsa"
	"crypto/elliptic"
	"crypto/rand"
	"crypto/tls"
	"crypto/x509"
	"crypto/x509/pkix"
	"encoding/pem"
	"fmt"
	"os"
	"path/filepath"
	"sync"
	"time"
)

// Store manages agent TLS certificates on disk.
type Store struct {
	dir        string
	mu         sync.RWMutex
	key        *ecdsa.PrivateKey
	cert       *x509.Certificate
	certDER    []byte
	caCertDER  []byte
	caCert     *x509.Certificate
}

// NewStore creates a cert store backed by the given directory.
func NewStore(dir string) *Store {
	return &Store{dir: dir}
}

// NewMemoryStore creates an in-memory cert store for testing.
func NewMemoryStore() *Store {
	return &Store{dir: ""}
}

// GenerateCSR creates a new ECDSA P-256 keypair and returns a DER-encoded CSR.
func (s *Store) GenerateCSR(serverID string) ([]byte, error) {
	key, err := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
	if err != nil {
		return nil, err
	}
	s.mu.Lock()
	s.key = key
	s.mu.Unlock()

	template := &x509.CertificateRequest{
		Subject: pkix.Name{CommonName: serverID},
	}
	return x509.CreateCertificateRequest(rand.Reader, template, key)
}

// StoreCert saves the signed certificate and CA cert.
func (s *Store) StoreCert(certDER, caCertDER []byte) error {
	cert, err := x509.ParseCertificate(certDER)
	if err != nil {
		return fmt.Errorf("parse cert: %w", err)
	}
	caCert, err := x509.ParseCertificate(caCertDER)
	if err != nil {
		return fmt.Errorf("parse CA cert: %w", err)
	}

	s.mu.Lock()
	s.cert = cert
	s.certDER = certDER
	s.caCert = caCert
	s.caCertDER = caCertDER
	s.mu.Unlock()

	if s.dir != "" {
		return s.writeToDisk()
	}
	return nil
}

// TLSConfig returns a tls.Config for mTLS if certs are available, nil otherwise.
func (s *Store) TLSConfig() *tls.Config {
	s.mu.RLock()
	defer s.mu.RUnlock()

	if s.key == nil || s.certDER == nil || s.caCertDER == nil {
		return nil
	}

	certPEM := pem.EncodeToMemory(&pem.Block{Type: "CERTIFICATE", Bytes: s.certDER})
	keyDER, _ := x509.MarshalECPrivateKey(s.key)
	keyPEM := pem.EncodeToMemory(&pem.Block{Type: "EC PRIVATE KEY", Bytes: keyDER})

	tlsCert, err := tls.X509KeyPair(certPEM, keyPEM)
	if err != nil {
		return nil
	}

	pool := x509.NewCertPool()
	pool.AddCert(s.caCert)

	return &tls.Config{
		Certificates: []tls.Certificate{tlsCert},
		RootCAs:      pool,
	}
}

// NeedsRenewal returns true if the cert expires within the given threshold.
func (s *Store) NeedsRenewal(threshold time.Duration) bool {
	s.mu.RLock()
	defer s.mu.RUnlock()
	if s.cert == nil {
		return false
	}
	return time.Until(s.cert.NotAfter) < threshold
}

// HasCert returns true if the store has a valid certificate.
func (s *Store) HasCert() bool {
	s.mu.RLock()
	defer s.mu.RUnlock()
	return s.cert != nil && s.key != nil
}

func (s *Store) writeToDisk() error {
	os.MkdirAll(s.dir, 0700)

	keyDER, err := x509.MarshalECPrivateKey(s.key)
	if err != nil {
		return err
	}

	files := map[string]*pem.Block{
		"client.crt": {Type: "CERTIFICATE", Bytes: s.certDER},
		"client.key": {Type: "EC PRIVATE KEY", Bytes: keyDER},
		"ca.crt":     {Type: "CERTIFICATE", Bytes: s.caCertDER},
	}

	for name, block := range files {
		path := filepath.Join(s.dir, name)
		data := pem.EncodeToMemory(block)
		if err := os.WriteFile(path, data, 0600); err != nil {
			return fmt.Errorf("write %s: %w", name, err)
		}
	}
	return nil
}

package server

import (
	"crypto/rand"
	"encoding/hex"
	"net/http"
	"strings"
	"time"
)

const (
	cookieAccess  = "aerodocs_access"
	cookieRefresh = "aerodocs_refresh"
	cookieCSRF    = "aerodocs_csrf"
)

func setAuthCookies(w http.ResponseWriter, accessToken, refreshToken string, isDev bool) {
	secure := !isDev

	http.SetCookie(w, &http.Cookie{
		Name:     cookieAccess,
		Value:    accessToken,
		Path:     "/",
		MaxAge:   900, // 15 minutes
		HttpOnly: true,
		Secure:   secure,
		SameSite: http.SameSiteStrictMode,
	})

	http.SetCookie(w, &http.Cookie{
		Name:     cookieRefresh,
		Value:    refreshToken,
		Path:     "/api/auth/refresh",
		MaxAge:   604800, // 7 days
		HttpOnly: true,
		Secure:   secure,
		SameSite: http.SameSiteStrictMode,
	})

	csrfToken := generateCSRFToken()
	http.SetCookie(w, &http.Cookie{
		Name:     cookieCSRF,
		Value:    csrfToken,
		Path:     "/",
		MaxAge:   604800,
		HttpOnly: false, // JS must read this
		Secure:   secure,
		SameSite: http.SameSiteStrictMode,
	})
}

func clearAuthCookies(w http.ResponseWriter) {
	for _, name := range []string{cookieAccess, cookieRefresh, cookieCSRF} {
		http.SetCookie(w, &http.Cookie{
			Name:   name,
			Value:  "",
			Path:   "/",
			MaxAge: -1,
		})
	}
}

func readAccessToken(r *http.Request) string {
	if c, err := r.Cookie(cookieAccess); err == nil && c.Value != "" {
		return c.Value
	}
	auth := r.Header.Get("Authorization")
	if strings.HasPrefix(auth, "Bearer ") {
		return strings.TrimPrefix(auth, "Bearer ")
	}
	return ""
}

func generateCSRFToken() string {
	b := make([]byte, 32)
	rand.Read(b)
	return hex.EncodeToString(b)
}

func readCSRFToken(r *http.Request) string {
	return r.Header.Get("X-CSRF-Token")
}

func readCSRFCookie(r *http.Request) string {
	if c, err := r.Cookie(cookieCSRF); err == nil {
		return c.Value
	}
	return ""
}

// isUsingBearerAuth returns true if the request uses Authorization header (not cookies).
func isUsingBearerAuth(r *http.Request) bool {
	return strings.HasPrefix(r.Header.Get("Authorization"), "Bearer ")
}

package server

import (
	"net/http"
)

func csrfMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		// Safe methods are exempt
		if r.Method == http.MethodGet || r.Method == http.MethodHead || r.Method == http.MethodOptions {
			next.ServeHTTP(w, r)
			return
		}

		// Bearer auth is exempt (non-browser clients)
		if isUsingBearerAuth(r) {
			next.ServeHTTP(w, r)
			return
		}

		// Validate CSRF token
		cookieToken := readCSRFCookie(r)
		headerToken := readCSRFToken(r)

		if cookieToken == "" || headerToken == "" || cookieToken != headerToken {
			respondError(w, http.StatusForbidden, "CSRF validation failed")
			return
		}

		next.ServeHTTP(w, r)
	})
}
```

export async function apiFetch<T>(
  path: string,
  options: RequestInit = {},
): Promise<T> {
  const headers = new Headers(options.headers)
  headers.set('X-CSRF-Token', getCSRFToken())

  let res = await fetch(`/api${path}`, {
    ...options,
    headers,
    credentials: 'same-origin',
  })

  if (res.status === 401) {
    // Try refresh — cookie sent automatically
    const refreshRes = await fetch('/api/auth/refresh', {
      method: 'POST',
      credentials: 'same-origin',
    })

    if (refreshRes.ok) {
      // Retry with new cookies (set by server)
      headers.set('X-CSRF-Token', getCSRFToken())
      res = await fetch(`/api${path}`, {
        ...options,
        headers,
        credentials: 'same-origin',
      })

      throw new Error('Session expired')
    }
  }

  if (!res.ok) {
