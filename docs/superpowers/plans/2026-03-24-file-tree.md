# File Tree Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build remote file browsing through the agent gRPC stream with file content viewing (syntax highlighting + Markdown rendering), and admin-configured path access control per user per server.

**Architecture:** Hub HTTP endpoints receive file browse/read requests, generate a request_id, forward them through the existing gRPC bidirectional stream to the agent, and wait for correlated responses. Agent handles incoming file requests via a new filebrowser package, returning directory listings and file content. The existing permissions table gates access per user per server per path. Frontend renders an IDE-like split view with a collapsible directory tree and a file viewer pane with syntax highlighting.

**Tech Stack:** Go (hub + agent), Protocol Buffers, React, TypeScript, TanStack Query, highlight.js, react-markdown

**Spec:** `docs/superpowers/specs/2026-03-24-file-tree-design.md`

// SendToAgent sends a message to a connected agent, holding the per-connection
// send mutex to prevent concurrent writes on the gRPC stream.
func (cm *ConnManager) SendToAgent(serverID string, msg *pb.HubMessage) error {
	conn := cm.GetConn(serverID)
	if conn == nil {
		return fmt.Errorf("agent not connected: %s", serverID)
	}
	conn.SendMu.Lock()
	defer conn.SendMu.Unlock()
	return conn.Stream.Send(msg)
}

```go
package grpcserver

import (
	"sync"

	"google.golang.org/protobuf/proto"
)

// PendingRequests tracks in-flight request-response correlations.
// An HTTP handler registers a channel keyed by request_id, sends the
// request through the agent stream, and waits on the channel for the
// agent's response. The gRPC receive loop delivers responses to the
// matching channel.
type PendingRequests struct {
	mu       sync.Mutex
	channels map[string]chan proto.Message
}

func NewPendingRequests() *PendingRequests {
	return &PendingRequests{
		channels: make(map[string]chan proto.Message),
	}
}

// Register creates a buffered channel for a request_id and returns it.
// The caller should defer Remove(requestID) to prevent leaks.
func (p *PendingRequests) Register(requestID string) chan proto.Message {
	ch := make(chan proto.Message, 1)
	p.mu.Lock()
	p.channels[requestID] = ch
	p.mu.Unlock()
	return ch
}

// Deliver sends a response to the waiting channel for the given request_id.
// Returns true if the channel was found and the message was delivered.
func (p *PendingRequests) Deliver(requestID string, msg proto.Message) bool {
	p.mu.Lock()
	ch, ok := p.channels[requestID]
	p.mu.Unlock()
	if !ok {
		return false
	}
	select {
	case ch <- msg:
		return true
	default:
		// Channel already has a message (shouldn't happen with buffered chan of 1)
		return false
	}
}

// Remove deletes a request_id from the map. Safe to call multiple times.
func (p *PendingRequests) Remove(requestID string) {
	p.mu.Lock()
	delete(p.channels, requestID)
	p.mu.Unlock()
}
```

func TestPendingRequests_RegisterAndDeliver(t *testing.T) {
	p := NewPendingRequests()
	ch := p.Register("req-1")
	defer p.Remove("req-1")

	resp := &pb.FileListResponse{RequestId: "req-1"}
	delivered := p.Deliver("req-1", resp)
	if !delivered {
		t.Fatal("expected delivery to succeed")
	}

	select {
	case msg := <-ch:
		got, ok := msg.(*pb.FileListResponse)
		if !ok {
			t.Fatalf("expected *FileListResponse, got %T", msg)
		}
		if got.RequestId != "req-1" {
			t.Fatalf("expected request_id 'req-1', got '%s'", got.RequestId)
		}
	case <-time.After(1 * time.Second):
		t.Fatal("timeout waiting for response")
	}
}

package filebrowser

import (
	"fmt"
	"io"
	"os"
	"path/filepath"
	"sort"
	"strings"

	pb "github.com/wyiu/aerodocs/proto/aerodocs/v1"
)

const maxReadBytes = 1048576 // 1MB

// validatePath ensures the path is absolute, clean, and contains no traversal.
func validatePath(path string) error {
	if !filepath.IsAbs(path) {
		return fmt.Errorf("path must be absolute: %s", path)
	}
	if strings.Contains(path, "..") {
		return fmt.Errorf("path traversal not allowed: %s", path)
	}
	return nil
}

// mimeFromExt returns a MIME type based on file extension.
func mimeFromExt(name string) string {
	ext := strings.ToLower(filepath.Ext(name))
	switch ext {
	case ".json":
		return "application/json"
	case ".xml":
		return "application/xml"
	case ".yaml", ".yml":
		return "application/x-yaml"
	case ".toml":
		return "application/toml"
	case ".md", ".markdown":
		return "text/markdown; charset=utf-8"
	case ".html", ".htm":
		return "text/html; charset=utf-8"
	case ".css":
		return "text/css; charset=utf-8"
	case ".js", ".mjs":
		return "text/javascript; charset=utf-8"
	case ".ts", ".tsx":
		return "text/typescript; charset=utf-8"
	case ".jsx":
		return "text/jsx; charset=utf-8"
	case ".go":
		return "text/x-go; charset=utf-8"
	case ".py":
		return "text/x-python; charset=utf-8"
	case ".rs":
		return "text/x-rust; charset=utf-8"
	case ".sh", ".bash":
		return "text/x-shellscript; charset=utf-8"
	case ".sql":
		return "text/x-sql; charset=utf-8"
	case ".proto":
		return "text/x-protobuf; charset=utf-8"
	case ".dockerfile":
		return "text/x-dockerfile; charset=utf-8"
	case ".csv":
		return "text/csv; charset=utf-8"
	case ".log", ".txt", ".text", ".conf", ".cfg", ".ini", ".env", ".properties":
		return "text/plain; charset=utf-8"
	default:
		return "application/octet-stream"
	}
}

// ListDir lists the contents of a directory, returning FileNode entries
// sorted with directories first, then files, both alphabetical.
func ListDir(path string) (*pb.FileListResponse, error) {
	if err := validatePath(path); err != nil {
		return nil, err
	}

	resolved, err := filepath.EvalSymlinks(path)
	if err != nil {
		return &pb.FileListResponse{
			Error: fmt.Sprintf("cannot resolve path: %v", err),
		}, nil
	}

	entries, err := os.ReadDir(resolved)
	if err != nil {
		return &pb.FileListResponse{
			Error: fmt.Sprintf("cannot read directory: %v", err),
		}, nil
	}

	var dirs, files []*pb.FileNode
	for _, entry := range entries {
		info, err := entry.Info()
		if err != nil {
			continue
		}

		entryPath := filepath.Join(path, entry.Name())
		node := &pb.FileNode{
			Name:  entry.Name(),
			Path:  entryPath,
			IsDir: entry.IsDir(),
			Size:  info.Size(),
		}

		// Check readability
		f, openErr := os.Open(filepath.Join(resolved, entry.Name()))
		if openErr == nil {
			node.Readable = true
			f.Close()
		}

		if entry.IsDir() {
			dirs = append(dirs, node)
		} else {
			files = append(files, node)
		}
	}

	// Sort each group alphabetically
	sort.Slice(dirs, func(i, j int) bool {
		return strings.ToLower(dirs[i].Name) < strings.ToLower(dirs[j].Name)
	})
	sort.Slice(files, func(i, j int) bool {
		return strings.ToLower(files[i].Name) < strings.ToLower(files[j].Name)
	})

	// Directories first, then files
	allNodes := append(dirs, files...)

	return &pb.FileListResponse{
		Files: allNodes,
	}, nil
}

// ReadFile reads a file from the filesystem with offset/limit support.
// limit must be <= maxReadBytes (1MB). Returns file content, total size,
// and detected MIME type.
func ReadFile(path string, offset, limit int64) (*pb.FileReadResponse, error) {
	if err := validatePath(path); err != nil {
		return nil, err
	}

	if limit > maxReadBytes {
		return nil, fmt.Errorf("limit %d exceeds maximum %d bytes", limit, maxReadBytes)
	}
	if limit <= 0 {
		limit = maxReadBytes
	}

	resolved, err := filepath.EvalSymlinks(path)
	if err != nil {
		return &pb.FileReadResponse{
			Error: fmt.Sprintf("cannot resolve path: %v", err),
		}, nil
	}

	info, err := os.Stat(resolved)
	if err != nil {
		return &pb.FileReadResponse{
			Error: fmt.Sprintf("cannot stat file: %v", err),
		}, nil
	}
	if info.IsDir() {
		return &pb.FileReadResponse{
			Error: "cannot read a directory",
		}, nil
	}

	f, err := os.Open(resolved)
	if err != nil {
		return &pb.FileReadResponse{
			Error: fmt.Sprintf("cannot open file: %v", err),
		}, nil
	}
	defer f.Close()

	if offset > 0 {
		if _, err := f.Seek(offset, io.SeekStart); err != nil {
			return &pb.FileReadResponse{
				Error: fmt.Sprintf("cannot seek: %v", err),
			}, nil
		}
	}

	data := make([]byte, limit)
	n, err := io.ReadFull(f, data)
	if err != nil && err != io.EOF && err != io.ErrUnexpectedEOF {
		return &pb.FileReadResponse{
			Error: fmt.Sprintf("cannot read file: %v", err),
		}, nil
	}
	data = data[:n]

	return &pb.FileReadResponse{
		Data:      data,
		TotalSize: info.Size(),
		MimeType:  mimeFromExt(path),
	}, nil
}

package client

import (
	"context"
	"fmt"
	"io"
	"log"
	"time"

	"google.golang.org/grpc"
	"google.golang.org/grpc/credentials/insecure"

	"github.com/wyiu/aerodocs/agent/internal/filebrowser"
	"github.com/wyiu/aerodocs/agent/internal/heartbeat"
	pb "github.com/wyiu/aerodocs/proto/aerodocs/v1"
)

type Config struct {
	HubAddr      string
	ServerID     string
	Token        string
	Hostname     string
	IPAddress    string
	OS           string
	AgentVersion string
}

type Client struct {
	hubAddr      string
	serverID     string
	token        string
	hostname     string
	ipAddress    string
	os           string
	agentVersion string
	backoff      time.Duration
	maxBackoff   time.Duration
}

func New(cfg Config) *Client {
	return &Client{
		hubAddr:      cfg.HubAddr,
		serverID:     cfg.ServerID,
		token:        cfg.Token,
		hostname:     cfg.Hostname,
		ipAddress:    cfg.IPAddress,
		os:           cfg.OS,
		agentVersion: cfg.AgentVersion,
		backoff:      1 * time.Second,
		maxBackoff:   60 * time.Second,
	}
}

func (c *Client) ServerID() string {
	return c.serverID
}

func (c *Client) Run(ctx context.Context) error {
	for {
		err := c.connectAndStream(ctx)
		if ctx.Err() != nil {
			return ctx.Err()
		}
		wait := c.nextBackoff()
		log.Printf("connection lost: %v — reconnecting in %v", err, wait)
		select {
		case <-ctx.Done():
			return ctx.Err()
		case <-time.After(wait):
		}
	}
}

func (c *Client) connectAndStream(ctx context.Context) error {
	conn, err := grpc.NewClient(c.hubAddr,
		grpc.WithTransportCredentials(insecure.NewCredentials()),
	)
	if err != nil {
		return fmt.Errorf("dial hub: %w", err)
	}
	defer conn.Close()

	client := pb.NewAgentServiceClient(conn)
	stream, err := client.Connect(ctx)
	if err != nil {
		return fmt.Errorf("open stream: %w", err)
	}

	if c.serverID == "" {
		if err := stream.Send(&pb.AgentMessage{
			Payload: &pb.AgentMessage_Register{
				Register: &pb.RegisterAgent{
					Token:        c.token,
					Hostname:     c.hostname,
					IpAddress:    c.ipAddress,
					Os:           c.os,
					AgentVersion: c.agentVersion,
				},
			},
		}); err != nil {
			return fmt.Errorf("send register: %w", err)
		}
		msg, err := stream.Recv()
		if err != nil {
			return fmt.Errorf("recv register ack: %w", err)
		}
		ack := msg.GetRegisterAck()
		if ack == nil {
			return fmt.Errorf("expected RegisterAck, got %T", msg.Payload)
		}
		if !ack.Success {
			return fmt.Errorf("registration rejected: %s", ack.Error)
		}
		c.serverID = ack.ServerId
		log.Printf("registered successfully: server_id=%s", c.serverID)
	} else {
		if err := stream.Send(heartbeat.BuildMessage(c.serverID)); err != nil {
			return fmt.Errorf("send heartbeat: %w", err)
		}
		msg, err := stream.Recv()
		if err != nil {
			return fmt.Errorf("recv heartbeat ack: %w", err)
		}
		if msg.GetHeartbeatAck() == nil {
			return fmt.Errorf("expected HeartbeatAck, got %T", msg.Payload)
		}
	}

	c.resetBackoff()
	log.Printf("connected to hub at %s", c.hubAddr)

	// Unified send channel -- all outbound messages go through here
	// so that only the main loop calls stream.Send (single writer).
	sendCh := make(chan *pb.AgentMessage, 16)

	// Heartbeat ticker pushes to sendCh
	hbStop := make(chan struct{})
	defer close(hbStop)
	go heartbeat.StartTicker(c.serverID, 10*time.Second, sendCh, hbStop)

	// Receive goroutine: reads from stream, dispatches requests, pushes responses to sendCh
	recvErr := make(chan error, 1)
	go func() {
		for {
			msg, err := stream.Recv()
			if err == io.EOF {
				recvErr <- nil
				return
			}
			if err != nil {
				recvErr <- err
				return
			}
			c.dispatchHubMessage(msg, sendCh)
		}
	}()

	// Main loop: drains sendCh to stream.Send (single writer)
	for {
		select {
		case <-ctx.Done():
			return ctx.Err()
		case err := <-recvErr:
			return fmt.Errorf("receive error: %w", err)
		case msg := <-sendCh:
			if err := stream.Send(msg); err != nil {
				return fmt.Errorf("send: %w", err)
			}
		}
	}
}

// dispatchHubMessage handles an incoming HubMessage from the Hub.
// It runs in the receive goroutine and must not call stream.Send directly.
// Instead, it pushes responses to sendCh.
func (c *Client) dispatchHubMessage(msg *pb.HubMessage, sendCh chan<- *pb.AgentMessage) {
	switch p := msg.Payload.(type) {
	case *pb.HubMessage_HeartbeatAck:
		// Already handled during initial handshake; ignore during stream
	case *pb.HubMessage_FileListRequest:
		go c.handleFileListRequest(p.FileListRequest, sendCh)
	case *pb.HubMessage_FileReadRequest:
		go c.handleFileReadRequest(p.FileReadRequest, sendCh)
	default:
		log.Printf("unhandled hub message: %T", p)
	}
}

func (c *Client) handleFileListRequest(req *pb.FileListRequest, sendCh chan<- *pb.AgentMessage) {
	resp, err := filebrowser.ListDir(req.Path)
	if err != nil {
		resp = &pb.FileListResponse{
			RequestId: req.RequestId,
			Error:     err.Error(),
		}
	} else {
		resp.RequestId = req.RequestId
	}
	sendCh <- &pb.AgentMessage{
		Payload: &pb.AgentMessage_FileListResponse{
			FileListResponse: resp,
		},
	}
}

func (c *Client) handleFileReadRequest(req *pb.FileReadRequest, sendCh chan<- *pb.AgentMessage) {
	resp, err := filebrowser.ReadFile(req.Path, req.Offset, req.Limit)
	if err != nil {
		resp = &pb.FileReadResponse{
			RequestId: req.RequestId,
			Error:     err.Error(),
		}
	} else {
		resp.RequestId = req.RequestId
	}
	sendCh <- &pb.AgentMessage{
		Payload: &pb.AgentMessage_FileReadResponse{
			FileReadResponse: resp,
		},
	}
}

func (c *Client) nextBackoff() time.Duration {
	current := c.backoff
	c.backoff *= 2
	if c.backoff > c.maxBackoff {
		c.backoff = c.maxBackoff
	}
	return current
}

func (c *Client) resetBackoff() {
	c.backoff = 1 * time.Second
}

// ListPermissionsForServer returns all permissions for a given server,
// joined with the users table to include usernames.
func (s *Store) ListPermissionsForServer(serverID string) ([]model.Permission, error) {
	rows, err := s.db.Query(
		`SELECT p.id, p.user_id, u.username, p.server_id, p.path, p.created_at
		 FROM permissions p
		 INNER JOIN users u ON p.user_id = u.id
		 WHERE p.server_id = ?
		 ORDER BY u.username, p.path`,
		serverID,
	)
	if err != nil {
		return nil, fmt.Errorf("list permissions: %w", err)
	}
	defer rows.Close()

	var perms []model.Permission
	for rows.Next() {
		var p model.Permission
		if err := rows.Scan(&p.ID, &p.UserID, &p.Username, &p.ServerID, &p.Path, &p.CreatedAt); err != nil {
			return nil, fmt.Errorf("scan permission: %w", err)
		}
		perms = append(perms, p)
	}
	return perms, rows.Err()
}

// GetUserPathsForServer returns the list of allowed paths for a user on a

package server

import (
	"encoding/base64"
	"fmt"
	"net/http"
	"path/filepath"
	"strings"
	"time"

	"github.com/google/uuid"
	"github.com/wyiu/aerodocs/hub/internal/model"
	pb "github.com/wyiu/aerodocs/proto/aerodocs/v1"
)

const (
	agentTimeout   = 10 * time.Second
	maxFileSize    = 10 * 1048576 // 10MB hard cap
	maxReadLimit   = 1048576      // 1MB per read
)

// cleanRequestPath validates and cleans a filesystem path from a query parameter.
func cleanRequestPath(path string) (string, error) {
	if path == "" {
		return "", fmt.Errorf("path is required")
	}
	if !filepath.IsAbs(path) {
		return "", fmt.Errorf("path must be absolute")
	}
	cleaned := filepath.Clean(path)
	if strings.Contains(path, "..") {
		return "", fmt.Errorf("path traversal not allowed")
	}
	return cleaned, nil
}

// isPathAllowed checks if a requested path is under one of the allowed root paths.
func isPathAllowed(requestedPath string, allowedPaths []string) bool {
	cleaned := filepath.Clean(requestedPath)
	if strings.Contains(requestedPath, "..") {
		return false
	}
	for _, allowed := range allowedPaths {
		allowedClean := filepath.Clean(allowed)
		if allowedClean == "/" {
			return true
		}
		if cleaned == allowedClean || strings.HasPrefix(cleaned, allowedClean+"/") {
			return true
		}
	}
	return false
}

// handleListFiles handles GET /api/servers/{id}/files?path=/var/log
func (s *Server) handleListFiles(w http.ResponseWriter, r *http.Request) {
	serverID := r.PathValue("id")
	requestedPath := r.URL.Query().Get("path")

	if requestedPath == "" {
		requestedPath = "/"
	}

	cleanedPath, err := cleanRequestPath(requestedPath)
	if err != nil {
		respondError(w, http.StatusBadRequest, err.Error())
		return
	}

	// Check server exists
	_, err = s.store.GetServerByID(serverID)
	if err != nil {
		respondError(w, http.StatusNotFound, "server not found")
		return
	}

	// Check path permissions
	role := UserRoleFromContext(r.Context())
	userID := UserIDFromContext(r.Context())

	if role != "admin" {
		allowedPaths, err := s.store.GetUserPathsForServer(userID, serverID)
		if err != nil {
			respondError(w, http.StatusInternalServerError, "failed to check permissions")
			return
		}
		if !isPathAllowed(cleanedPath, allowedPaths) {
			respondError(w, http.StatusForbidden, "access denied")
			return
		}
	}

	// Check agent is connected
	if s.connMgr == nil {
		respondError(w, http.StatusBadGateway, "agent not connected")
		return
	}
	conn := s.connMgr.GetConn(serverID)
	if conn == nil {
		respondError(w, http.StatusBadGateway, "agent not connected")
		return
	}

	// Send request through the stream
	requestID := uuid.NewString()
	ch := s.pending.Register(requestID)
	defer s.pending.Remove(requestID)

	err = s.connMgr.SendToAgent(serverID, &pb.HubMessage{
		Payload: &pb.HubMessage_FileListRequest{
			FileListRequest: &pb.FileListRequest{
				RequestId: requestID,
				Path:      cleanedPath,
			},
		},
	})
	if err != nil {
		respondError(w, http.StatusBadGateway, "failed to send request to agent")
		return
	}

	// Wait for response with timeout
	select {
	case msg := <-ch:
		resp, ok := msg.(*pb.FileListResponse)
		if !ok {
			respondError(w, http.StatusInternalServerError, "unexpected response type")
			return
		}
		if resp.Error != "" {
			respondError(w, http.StatusNotFound, resp.Error)
			return
		}
		// Convert to JSON-friendly format
		type fileNode struct {
			Name     string `json:"name"`
			Path     string `json:"path"`
			IsDir    bool   `json:"is_dir"`
			Size     int64  `json:"size"`
			Readable bool   `json:"readable"`
		}
		nodes := make([]fileNode, 0, len(resp.Files))
		for _, f := range resp.Files {
			nodes = append(nodes, fileNode{
				Name:     f.Name,
				Path:     f.Path,
				IsDir:    f.IsDir,
				Size:     f.Size,
				Readable: f.Readable,
			})
		}
		respondJSON(w, http.StatusOK, map[string]interface{}{
			"files": nodes,
		})
	case <-time.After(agentTimeout):
		respondError(w, http.StatusGatewayTimeout, "agent timeout")
	}
}

// handleReadFile handles GET /api/servers/{id}/files/read?path=/var/log/syslog
func (s *Server) handleReadFile(w http.ResponseWriter, r *http.Request) {
	serverID := r.PathValue("id")
	requestedPath := r.URL.Query().Get("path")

	cleanedPath, err := cleanRequestPath(requestedPath)
	if err != nil {
		respondError(w, http.StatusBadRequest, err.Error())
		return
	}

	// Check server exists
	_, err = s.store.GetServerByID(serverID)
	if err != nil {
		respondError(w, http.StatusNotFound, "server not found")
		return
	}

	// Check path permissions
	role := UserRoleFromContext(r.Context())
	userID := UserIDFromContext(r.Context())

	if role != "admin" {
		allowedPaths, err := s.store.GetUserPathsForServer(userID, serverID)
		if err != nil {
			respondError(w, http.StatusInternalServerError, "failed to check permissions")
			return
		}
		if !isPathAllowed(cleanedPath, allowedPaths) {
			respondError(w, http.StatusForbidden, "access denied")
			return
		}
	}

	// Check agent is connected
	if s.connMgr == nil {
		respondError(w, http.StatusBadGateway, "agent not connected")
		return
	}
	conn := s.connMgr.GetConn(serverID)
	if conn == nil {
		respondError(w, http.StatusBadGateway, "agent not connected")
		return
	}

	// Send request through the stream
	requestID := uuid.NewString()
	ch := s.pending.Register(requestID)
	defer s.pending.Remove(requestID)

	err = s.connMgr.SendToAgent(serverID, &pb.HubMessage{
		Payload: &pb.HubMessage_FileReadRequest{
			FileReadRequest: &pb.FileReadRequest{
				RequestId: requestID,
				Path:      cleanedPath,
				Offset:    0,
				Limit:     maxReadLimit,
			},
		},
	})
	if err != nil {
		respondError(w, http.StatusBadGateway, "failed to send request to agent")
		return
	}

	// Wait for response with timeout
	select {
	case msg := <-ch:
		resp, ok := msg.(*pb.FileReadResponse)
		if !ok {
			respondError(w, http.StatusInternalServerError, "unexpected response type")
			return
		}
		if resp.Error != "" {
			if strings.Contains(resp.Error, "cannot open file") {
				respondError(w, http.StatusForbidden, resp.Error)
			} else {
				respondError(w, http.StatusNotFound, resp.Error)
			}
			return
		}

		// Hard cap check
		if resp.TotalSize > maxFileSize {
			respondError(w, http.StatusRequestEntityTooLarge, fmt.Sprintf("file too large: %d bytes (max %d)", resp.TotalSize, maxFileSize))
			return
		}

		// If file is larger than 1MB, re-request the last 1MB
		if resp.TotalSize > maxReadLimit {
			// Send a new request for the tail
			tailRequestID := uuid.NewString()
			tailCh := s.pending.Register(tailRequestID)
			defer s.pending.Remove(tailRequestID)

			tailOffset := resp.TotalSize - maxReadLimit
			err = s.connMgr.SendToAgent(serverID, &pb.HubMessage{
				Payload: &pb.HubMessage_FileReadRequest{
					FileReadRequest: &pb.FileReadRequest{
						RequestId: tailRequestID,
						Path:      cleanedPath,
						Offset:    tailOffset,
						Limit:     maxReadLimit,
					},
				},
			})
			if err != nil {
				respondError(w, http.StatusBadGateway, "failed to send tail request to agent")
				return
			}

			select {
			case tailMsg := <-tailCh:
				tailResp, ok := tailMsg.(*pb.FileReadResponse)
				if !ok {
					respondError(w, http.StatusInternalServerError, "unexpected response type")
					return
				}
				if tailResp.Error != "" {
					respondError(w, http.StatusInternalServerError, tailResp.Error)
					return
				}
				resp = tailResp
			case <-time.After(agentTimeout):
				respondError(w, http.StatusGatewayTimeout, "agent timeout (tail read)")
				return
			}
		}

		// Log audit
		ip := clientIP(r)
		s.store.LogAudit(model.AuditEntry{
			ID: uuid.NewString(), UserID: &userID,
			Action: model.AuditFileRead, Target: &serverID,
			Detail: &cleanedPath, IPAddress: &ip,
		})

		respondJSON(w, http.StatusOK, map[string]interface{}{
			"data":       base64.StdEncoding.EncodeToString(resp.Data),
			"total_size": resp.TotalSize,
			"mime_type":  resp.MimeType,
		})
	case <-time.After(agentTimeout):
		respondError(w, http.StatusGatewayTimeout, "agent timeout")
	}
}

// handleCreatePath handles POST /api/servers/{id}/paths (admin only)
func (s *Server) handleCreatePath(w http.ResponseWriter, r *http.Request) {
	serverID := r.PathValue("id")

	var req struct {
		UserID string `json:"user_id"`
		Path   string `json:"path"`
	}
	if err := decodeJSON(r, &req); err != nil {
		respondError(w, http.StatusBadRequest, "invalid request body")
		return
	}

	if req.UserID == "" {
		respondError(w, http.StatusBadRequest, "user_id is required")
		return
	}
	if req.Path == "" {
		respondError(w, http.StatusBadRequest, "path is required")
		return
	}

	// Validate the path format
	cleanedPath, err := cleanRequestPath(req.Path)
	if err != nil {
		respondError(w, http.StatusBadRequest, err.Error())
		return
	}

	// Verify server exists
	_, err = s.store.GetServerByID(serverID)
	if err != nil {
		respondError(w, http.StatusNotFound, "server not found")
		return
	}

	id, err := s.store.CreatePermission(req.UserID, serverID, cleanedPath)
	if err != nil {
		respondError(w, http.StatusConflict, "permission already exists or invalid reference")
		return
	}

	// Audit log
	adminID := UserIDFromContext(r.Context())
	ip := clientIP(r)
	detail := req.UserID + " -> " + cleanedPath
	s.store.LogAudit(model.AuditEntry{
		ID: uuid.NewString(), UserID: &adminID,
		Action: model.AuditPathGranted, Target: &serverID,
		Detail: &detail, IPAddress: &ip,
	})

	respondJSON(w, http.StatusCreated, map[string]interface{}{
		"id":      id,
		"user_id": req.UserID,
		"path":    cleanedPath,
	})
}

// handleDeletePath handles DELETE /api/servers/{id}/paths/{pathId} (admin only)
func (s *Server) handleDeletePath(w http.ResponseWriter, r *http.Request) {
	serverID := r.PathValue("id")
	pathID := r.PathValue("pathId")

	perm, err := s.store.GetPermissionByID(pathID)

		return
	}

function formatBytes(bytes: number): string {
  if (bytes === 0) return '0 B'
  const k = 1024
  const sizes = ['B', 'KB', 'MB', 'GB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i]
}

function relativeTime(dateStr: string | null): string {
  if (!dateStr) return 'Never'
  const date = new Date(dateStr)
  const now = new Date()
  const diffMs = now.getTime() - date.getTime()
  const diffSec = Math.floor(diffMs / 1000)
  if (diffSec < 60) return `${diffSec}s ago`
  const diffMin = Math.floor(diffSec / 60)
  if (diffMin < 60) return `${diffMin} min ago`
  const diffHr = Math.floor(diffMin / 60)
  if (diffHr < 24) return `${diffHr}h ago`
  const diffDay = Math.floor(diffHr / 24)
  return `${diffDay}d ago`
}

          {/* Left Sidebar: Tree */}
          <div className="w-64 border-r border-border overflow-y-auto bg-surface/30 shrink-0">
            <div className="px-3 py-2 text-[10px] text-text-faint uppercase tracking-widest border-b border-border">
              File Tree
            </div>
            {allowedPaths.length === 0 ? (
              <div className="px-3 py-4 text-text-muted text-xs text-center">
                No paths configured. Ask an admin to grant access.
              </div>
            ) : (
              <div className="py-1">
                {rootNodes.map((node) => (
                  <TreeNode
                    key={node.path}
                    node={node}
                    level={0}
                    serverID={serverID}
                    selectedPath={selectedFile}
                    expandedDirs={expandedDirs}
                    loadingDirs={loadingDirs}
                    childrenMap={childrenMap}
                    onToggleDir={handleToggleDir}
                    onSelectFile={handleSelectFile}
                  />
                ))}
              </div>
            )}
          </div>

          {/* Right Pane: File Viewer */}
          <div className="flex-1 flex flex-col overflow-hidden">
            {!selectedFile ? (
              <div className="flex-1 flex items-center justify-center text-text-muted text-sm">
                Select a file to view its contents
              </div>
            ) : (
              <>
                {/* Viewer Header */}
                <div className="flex items-center justify-between px-4 py-2 bg-surface border-b border-border shrink-0">
                  <div className="flex items-center gap-2 text-xs text-text-muted overflow-hidden">
                    <ChevronLeft className="w-3 h-3 shrink-0" />
                    {breadcrumbs.map((part, i) => (
                      <span key={i} className="flex items-center gap-1 shrink-0">
                        {i > 0 && <span className="text-text-faint">/</span>}
                        <span
                          className={
                            i === breadcrumbs.length - 1
                              ? 'text-text-primary'
                              : ''
                          }
                        >
                          {part}
                        </span>
                      </span>
                    ))}
                  </div>
                  <div className="flex items-center gap-2 shrink-0">
                    {fileContent && (
                      <span className="text-[10px] text-text-faint">
                        {formatBytes(fileContent.total_size)}
                        {fileContent.mime_type && ` / ${fileContent.mime_type.split(';')[0]}`}
                      </span>
                    )}
                    {isMarkdown && (
                      <button
                        onClick={() => setShowRawMarkdown(!showRawMarkdown)}
                        className="px-2 py-0.5 text-[10px] bg-elevated text-text-muted hover:text-text-secondary rounded transition-colors"
                      >
                        {showRawMarkdown ? 'Rendered' : 'Raw'}
                      </button>
                    )}
                    <button
                      onClick={() => refetchFile()}
                      className="p-1 text-text-muted hover:text-text-secondary transition-colors"
                      title="Refresh"
                    >
                      <RefreshCw className="w-3.5 h-3.5" />
                    </button>
                  </div>
                </div>

                {/* Large file banner */}
                {isLargeFile && (
                  <div className="px-4 py-1.5 bg-elevated border-b border-border text-[10px] text-status-warning">
                    Showing last 1 MB of {formatBytes(fileContent.total_size)}
                  </div>
                )}

                {/* Content */}
                <div className="flex-1 overflow-auto">
                  {isFileLoading ? (
                    <div className="flex items-center justify-center py-8">
                      <RefreshCw className="w-4 h-4 text-text-faint animate-spin" />
                    </div>
                  ) : fileError ? (
                    <div className="px-4 py-8 text-center">
                      <p className="text-status-error text-sm mb-2">
                        {(fileError as Error).message}
                      </p>
                      <button
                        onClick={() => refetchFile()}
                        className="text-accent text-xs hover:underline"
                      >
                        Try again
                      </button>
                    </div>
                  ) : isMarkdown && !showRawMarkdown ? (
                    <MarkdownRenderer content={decodedContent} />
                  ) : (
                    <pre className="p-4 text-xs font-mono leading-relaxed overflow-x-auto">
                      {/* highlight.js output is sanitized code tokens, not user HTML */}
                      <code
                        dangerouslySetInnerHTML={{ __html: highlightedCode || escapeHtml(decodedContent) }}
                      />
                    </pre>
                  )}
                </div>
              </>
            )}
          </div>
        </div>
      )}

      {/* Path Management (admin only) */}
      {isAdmin && server && (
        <div className="border-t border-border bg-surface shrink-0">
          <button
            onClick={() => setShowPathManager(!showPathManager)}
            className="w-full flex items-center gap-2 px-4 py-2 text-xs text-text-muted hover:text-text-secondary transition-colors"
          >
            {showPathManager ? (
              <ChevronDown className="w-3 h-3" />
            ) : (
              <ChevronRight className="w-3 h-3" />
            )}
            Manage File Access
          </button>
          {showPathManager && (
            <PathManager serverID={serverID} />
          )}
        </div>
      )}
    </div>
  )
}

// -- HTML escape for fallback (when highlight.js fails) --

function escapeHtml(text: string): string {
  return text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
}

// -- Markdown Renderer (lazy loaded) --

function MarkdownRenderer({ content }: { content: string }) {
  const [ReactMarkdown, setReactMarkdown] = useState<React.ComponentType<{
    children: string
    remarkPlugins?: unknown[]
  }> | null>(null)
  const [remarkGfm, setRemarkGfm] = useState<unknown>(null)

  useEffect(() => {
    Promise.all([
      import('react-markdown'),
      import('remark-gfm'),
    ]).then(([md, gfm]) => {
      setReactMarkdown(() => md.default)
      setRemarkGfm(() => gfm.default)
    })
  }, [])

  if (!ReactMarkdown) {
    return <div className="p-4 text-text-muted text-sm">Loading renderer...</div>
  }

  return (
    <div className="p-4 prose prose-invert prose-sm max-w-none [&_pre]:bg-elevated [&_pre]:p-3 [&_pre]:rounded [&_code]:text-xs [&_code]:font-mono [&_h1]:text-text-primary [&_h2]:text-text-primary [&_h3]:text-text-primary [&_p]:text-text-secondary [&_li]:text-text-secondary [&_a]:text-accent [&_table]:border-border [&_th]:border-border [&_td]:border-border [&_hr]:border-border [&_blockquote]:border-accent [&_blockquote]:text-text-muted">
      <ReactMarkdown remarkPlugins={remarkGfm ? [remarkGfm] : []}>
        {content}
      </ReactMarkdown>
    </div>
  )
}

// -- Path Manager Component (admin only) --

function PathManager({ serverID }: { serverID: string }) {
  const queryClient = useQueryClient()
  const [newUserID, setNewUserID] = useState('')
  const [newPath, setNewPath] = useState('')

  // List all permissions for this server
  const { data: permsData } = useQuery({
    queryKey: ['server-permissions', serverID],
    queryFn: () => apiFetch<PathListResponse>(`/servers/${serverID}/paths`),
  })

  // List users for the dropdown
  const { data: usersData } = useQuery({
    queryKey: ['users'],
    queryFn: () => apiFetch<{ users: User[] }>('/users'),
  })

  const permissions = permsData?.paths ?? []
  const users = usersData?.users ?? []

  const createMutation = useMutation({
    mutationFn: () =>
      apiFetch(`/servers/${serverID}/paths`, {
        method: 'POST',
        body: JSON.stringify({ user_id: newUserID, path: newPath }),
      }),
    onSuccess: () => {
      setNewUserID('')
      setNewPath('')
      queryClient.invalidateQueries({ queryKey: ['server-permissions', serverID] })
    },
  })

  const deleteMutation = useMutation({
    mutationFn: (pathId: string) =>
      apiFetch(`/servers/${serverID}/paths/${pathId}`, { method: 'DELETE' }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['server-permissions', serverID] })
    },
  })

  return (
    <div className="px-4 pb-4">
      {/* Existing permissions table */}
      {permissions.length === 0 ? (
        <p className="text-text-muted text-xs mb-3">No path permissions configured.</p>
      ) : (
        <div className="border border-border rounded mb-3 overflow-hidden">
          <table className="w-full text-xs">
            <thead>
              <tr className="bg-elevated text-text-muted uppercase tracking-wider text-[10px]">
                <th className="px-3 py-1.5 text-left">User</th>
                <th className="px-3 py-1.5 text-left">Path</th>
                <th className="px-3 py-1.5 text-right">Actions</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-border">
              {permissions.map((perm: PathPermission) => (
                <tr key={perm.id} className="hover:bg-elevated/50">
                  <td className="px-3 py-1.5 text-text-secondary">{perm.username}</td>
                  <td className="px-3 py-1.5 font-mono text-text-muted">{perm.path}</td>
                  <td className="px-3 py-1.5 text-right">
                    <button
                      onClick={() => deleteMutation.mutate(perm.id)}
                      disabled={deleteMutation.isPending}
                      className="text-text-muted hover:text-status-offline transition-colors"
                    >
                      <Trash2 className="w-3 h-3" />
                    </button>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}

      {/* Add new permission */}
      <div className="flex items-center gap-2">
        <select
          value={newUserID}
          onChange={(e) => setNewUserID(e.target.value)}
          className="px-2 py-1 bg-elevated border border-border rounded text-xs text-text-primary focus:outline-none focus:border-accent"
        >
          <option value="">Select user...</option>
          {users.map((u: User) => (
            <option key={u.id} value={u.id}>
              {u.username}
            </option>
          ))}
        </select>
        <input
          type="text"
          value={newPath}
          onChange={(e) => setNewPath(e.target.value)}
          placeholder="/var/log"
          className="px-2 py-1 bg-elevated border border-border rounded text-xs text-text-primary placeholder:text-text-faint focus:outline-none focus:border-accent flex-1 max-w-xs font-mono"
        />
        <button
          onClick={() => createMutation.mutate()}
          disabled={!newUserID || !newPath || createMutation.isPending}
          className="flex items-center gap-1 px-2 py-1 bg-accent hover:bg-accent-hover text-white text-xs rounded transition-colors disabled:opacity-50"
        >
          <Plus className="w-3 h-3" />
          Add
        </button>
      </div>
      {createMutation.isError && (
        <p className="text-status-error text-xs mt-1">
          {(createMutation.error as Error).message}
        </p>
      )}
    </div>
  )
}
